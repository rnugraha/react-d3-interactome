## Install

```bash
npm install --save react-d3-interactome
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-d3-interactome'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
